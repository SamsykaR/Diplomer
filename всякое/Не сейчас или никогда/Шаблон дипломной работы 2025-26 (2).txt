    11.     QString severityString;
    12.     switch (severity) {

    14.         case Info:    severityString = "Info";    break;
    15.         case Warning: severityString = "Warning"; break;
    16.         case Error:   severityString = "Error";   break;
    17.         case Fatal:   severityString = "Fatal";   break;
    18.     }
